# Multi-platform order export — Shopee, Lazada, TikTok Shop
import pandas as pd
import glob, os
 
def merge_orders(folder):
    files = glob.glob(f"{folder}/*.xlsx")
    frames = []
    for f in files:
        platform = os.path.basename(f).split("_")[0]
        df = pd.read_excel(f)
        df["Platform"] = platform
        frames.append(df)
    merged = pd.concat(frames, ignore_index=True)
    merged.to_excel("combined_orders.xlsx", index=False)
    print(f"✅ {len(merged)} orders merged from {len(files)} platforms")

# Web scraper — track competitor prices daily
from bs4 import BeautifulSoup
import requests, pandas as pd
 
def scrape_prices(urls):
    results = []
    for url in urls:
        soup = BeautifulSoup(requests.get(url).text)
        price = soup.find("span", class_="price").text
        results.append({"url": url, "price": price})
    return pd.DataFrame(results)
